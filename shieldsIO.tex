\def\shield#1#2#3#4#5#6{
    \definecolor{LHScolor}{HTML}{#1}
    \definecolor{RHScolor}{HTML}{#3}
    \tcbsidebyside[
        bicolor,
        sidebyside,
        sidebyside adapt=both,
        sidebyside gap=5pt,
        top=0pt,left=0pt,right=0pt,bottom=0pt,
        boxrule=0pt,rounded corners,
        interior style={top color=LHScolor,bottom color=LHScolor!60!black},
        segmentation style={top color=RHScolor,bottom color=RHScolor!60!black},
    ]{%
        \href{#5}{\color{white}#2}%
    }{%
        \href{#6}{\color{white}#4}%
    }%
}
